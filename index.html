Objects

Everything is one


false.toString(); // 'false'
[1, 2, 3].toString(); // '1,2,3'

function Foo(){}
Foo.bar = 1;
Foo.bar; // 1

Even numbers


2.toString(); // raises SyntaxError

2..toString(); // the second point is correctly recognized
2 .toString(); // note the space left to the dot
(2).toString(); // 2 is evaluated first

But everyone uses them like a hash


var foo = {}; // a new empty object

// a new object with a 'test' property with value 12
var bar = {test: 12};

With every hash key access mechanism possible


var foo = {name: 'kitten'}
foo.name; // kitten
foo['name']; // kitten

var get = 'name';
foo[get]; // kitten

foo.1234; // SyntaxError
foo['1234']; // works

But inheritance is different from anything you're used to


function Foo() {
    this.value = 42;
}
Foo.prototype = {
    method: function() {}
};

function Bar() {}

// Set Bar's prototype to a new instance of Foo
Bar.prototype = new Foo();
Bar.prototype.foo = 'Hello World';

// Make sure to list Bar as the actual constructor
Bar.prototype.constructor = Bar;

var test = new Bar(); // create a new bar instance

// The resulting prototype chain
test [instance of Bar]
    Bar.prototype [instance of Foo]
        { foo: 'Hello World' }
        Foo.prototype
            { method: ... }
            Object.prototype
                { toString: ... /* etc. */ }

The Funky Stuff

'this this' or 'that this'

By default 'this' is global


execute = function();
this.execute()
//is the same as
execute()

Unless the function is in an object


value = "foo";
test = {
  value: "bar",
  execute: function() { return this.value }
};
test.execute()
// returns "bar"

Or other...reasons


value = "foo";
test = {
  value: "bar",
  execute: function() {
    function sub(){
      return this.value;
    }
    return sub();
  }
};
test.execute()
// returns "foo"

So people get it around it by explicitely setting it


value = "foo";
test = {
  value: "bar",
  execute: function() {
    var that = this;
    function sub(){
      return that.value;
    }
    return sub();
  }
};
test.execute()
// returns "bar"

Or by calling the method differently


value = "foo";
test = {
  value: "bar",
  execute: function() {
    function sub(){
      return this.value;
    }
    return sub.call(this);
  }
};
test.execute()
// returns "bar"

Types

Equality


""           ==   "0"           // false
0            ==   ""            // true
0            ==   "0"           // true
false        ==   "false"       // false
false        ==   "0"           // true
false        ==   undefined     // false
false        ==   null          // false
null         ==   undefined     // true
" \t\r\n"    ==   0             // true

is broken

Which why everyone uses triple equals


""           ===   "0"           // false
0            ===   ""            // false
0            ===   "0"           // false
false        ===   "false"       // false
false        ===   "0"           // false
false        ===   undefined     // false
false        ===   null          // false
null         ===   undefined     // false
" \t\r\n"    ===   0             // false

is broken

Programmatically determining types is an even bigger mess


Value               Class      Type
-------------------------------------
"foo"               String     string
new String("foo")   String     object
1.2                 Number     number
new Number(1.2)     Number     object
true                Boolean    boolean
new Boolean(true)   Boolean    object
new Date()          Date       object
new Error()         Error      object
[1,2,3]             Array      object
new Array(1, 2, 3)  Array      object
new Function("")    Function   function
/abc/g              RegExp     object (function in Nitro/V8)
new RegExp("meow")  RegExp     object (function in Nitro/V8)
{}                  Object     object
new Object()        Object     object

So use a library like lodash


_.isNumber(8.4);
// → true

_.isNumber(NaN);
// → true

_.isNumber('8.4');
// → false

Iteration is also kind of broken


// Poisoning Object.prototype
Object.prototype.bar = 1;

var foo = {moo: 2};
for(var i in foo) {
    console.log(i); // prints both bar and moo
}

So use a library like lodash


_([1, 2]).forEach(function(n) {
  console.log(n);
}).value();
// → logs each value from left to right and returns the array

_.forEach({ 'a': 1, 'b': 2 }, function(n, key) {
  console.log(n, key);
});
// → logs each value-key pair and returns the object (iteration order is not guaranteed)

The (kind of) Advanced Stuff

Controlling Scope

Remember this?


// global scope
var foo = 42;
function test() {
  // local scope
  foo = 21;
}
test();
foo; // 21

Everything not in an object or function is global by default!

We can fix that by putting everything in a function


(function(){
  // local scope
  var foo = 42;
  function test() {
    // local scope
    foo = 21;
  }
  test();
  foo; // 21
})();

Or using something like Browserify


module.exports = function(){
  // local scope
  var foo = 42;
  function test() {
    // local scope
    foo = 21;
  }
  test();
  foo; // 21
};
